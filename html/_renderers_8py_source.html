556  def build_cylinder(self, n_phis):
557  """Return the id of a display list that draws a cylinder.
558 
559  The cylinder is along the x axis from -1/2 to 1/2. It has a
560  circular cross-section with radius 1.
561 
562  n_phis -- The number of rectangular panels with which to
563  approximate the cylinder.
564  """
565  index = glGenLists(1)
566  phis = [float(i)*2.0*numpy.pi/float(n_phis) for i in range(n_phis+1)]
567  phi_pairs = zip(phis, phis[1:])
568  glNewList(index, GL_COMPILE)
569  glBegin(GL_QUADS)
570  for phi1,phi2 in phi_pairs:
571  dot1 = min(max(numpy.cos(phi1), 0.0), 1.0)
572  dot2 = min(max(numpy.cos(phi2), 0.0), 1.0)
573  glTexCoord1f(dot1)
574  glVertex3f(-0.5, numpy.sin(phi1), numpy.cos(phi1))
575  glTexCoord1f(dot1)
576  glVertex3f(0.5, numpy.sin(phi1), numpy.cos(phi1))
577  glTexCoord1f(dot2)
578  glVertex3f(0.5, numpy.sin(phi2), numpy.cos(phi2))
579  glTexCoord1f(dot2)
580  glVertex3f(-0.5, numpy.sin(phi2), numpy.cos(phi2))

585  def build_hemisphere(self, n_phis, n_thetas):
586  """Return the id of a display list that draw a hemisphere.
587 
588  The hemisphere is in the +x half of the x axis and opens on the x=0 plane. It has a radius of 1.
589 
590  n_phis -- Number of latitudes to use. There are actually n_phis/2+1 latitudes.
591  n_thetas -- Number of longitudes to use.
592  """
593  index = glGenLists(1)
594  phis = [float(i)*numpy.pi/float(n_phis/2) for i in range(n_phis/2+1)]
595  phi_pairs = zip(phis, phis[1:])
596  thetas = [float(i)*numpy.pi/float(n_thetas) for i in range(-n_thetas/2, n_thetas/2+1)]
597  theta_pairs = zip(thetas, thetas[1:])
598  glNewList(index, GL_COMPILE)
599  glBegin(GL_QUADS)
600  for phi1,phi2 in phi_pairs:
601  dot1 = min(max(numpy.cos(phi1), 0.0), 1.0)
602  dot2 = min(max(numpy.cos(phi2), 0.0), 1.0)
603  for th1,th2 in theta_pairs:
604  glTexCoord1f(dot1)
605  glVertex3f(numpy.sin(phi1)*numpy.cos(th1), numpy.sin(phi1)*numpy.sin(th1), numpy.cos(phi1))
606  glTexCoord1f(dot2)
607  glVertex3f(numpy.sin(phi2)*numpy.cos(th1), numpy.sin(phi2)*numpy.sin(th1), numpy.cos(phi2))
608  glTexCoord1f(dot2)
609  glVertex3f(numpy.sin(phi2)*numpy.cos(th2), numpy.sin(phi2)*numpy.sin(th2), numpy.cos(phi2))
610  glTexCoord1f(dot1)
611  glVertex3f(numpy.sin(phi1)*numpy.cos(th2), numpy.sin(phi1)*numpy.sin(th2), numpy.cos(phi1))
612  glEnd()
613  glEndList()
614  return index
